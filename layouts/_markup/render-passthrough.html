{{- /* Overrides the hextra render hook to render math leniently. The theme renders KaTeX server-side in strict mode and calls errorf on any markup KaTeX rejects, which fails the whole build. Our database has some LaTeX (e.g. {array} matrices) that KaTeX's strict mode flags. We pass strict=ignore and throwOnError=false so such snippets still render (KaTeX shows its own inline message at worst) and downgrade the hook's own failure to a warning so the site always builds. */ -}} {{- $engine := site.Params.math.engine | default "katex" -}} {{- if eq $engine "katex" -}} {{- $opts := dict "output" "htmlAndMathml" "displayMode" (eq .Type "block") "throwOnError" false "strict" "ignore" -}} {{- with try (transform.ToMath .Inner $opts) }} {{- with .Err }} {{- warnf "KaTeX could not render math at %s: %s" $.Position . -}} {{- else }} {{- .Value }} {{- $.Page.Store.Set "hasMath" true }} {{- end }} {{- end }} {{- else -}} {{/* MathJax - need to add delimiters back in */}} {{- $.Page.Store.Set "hasMath" true }} {{- if eq .Type "block" -}} \[{{- .Inner -}}\] {{- else -}} \( {{- .Inner -}} \) {{- end -}} {{- end -}}
